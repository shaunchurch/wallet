
WalletResponse dapp variants (popup receives these):
```typescript
| { type: 'dapp:pendingRequest'; request: PendingDappRequest | null }
| { type: 'dapp:approved' }
| { type: 'dapp:rejected' }
| { type: 'dapp:txSent'; txHash: string }
| { type: 'dapp:signed'; signature: string }
| { type: 'dapp:simulated'; ethBefore: string; ethAfter: string; success: boolean; error?: string }
```

```typescript
{
  id: string;
  rpcId: number;
  method: string;        // eth_requestAccounts | eth_sendTransaction | personal_sign | eth_signTypedData_v4
  params?: unknown[];
  origin: string;
  favicon?: string;
  title?: string;
  createdAt: number;
}
```
