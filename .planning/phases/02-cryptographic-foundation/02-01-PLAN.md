```typescript
// Vault blob stored in chrome.storage.local
export interface VaultBlob {
  version: 1;
  kdf: {
    algorithm: 'PBKDF2';
    hash: 'SHA-256';
    iterations: number;    // 600_000+
    salt: string;          // hex-encoded, 16 bytes random
  };
  cipher: {
    algorithm: 'AES-256-GCM';
    iv: string;            // hex-encoded, 12 bytes random
  };
  data: string;            // hex-encoded ciphertext (includes GCM auth tag)
}

// Plaintext stored inside encrypted vault
export interface VaultPlaintext {
  mnemonic: string;        // BIP-39 mnemonic phrase
  createdAt: number;       // unix timestamp ms
}

// A derived account (public info only -- safe to pass outside background)
export interface DerivedAccount {
  index: number;           // BIP-44 address_index
  address: string;         // EIP-55 checksummed 0x address
  path: string;            // full derivation path
}

// Internal key pair (NEVER leaves background service worker)
export interface KeyPair {
  privateKey: Uint8Array;  // 32 bytes
  address: string;         // EIP-55 checksummed
  path: string;            // derivation path
}

// Message types -- popup sends to background
export type WalletMessage =
  | { type: 'wallet:create'; password: string; strength?: 128 | 256 }
  | { type: 'wallet:confirmSeedPhrase'; wordIndices: Array<{ position: number; word: string }> }  // SEC-03: proof-of-confirmation — caller provides word at each challenged position
  | { type: 'wallet:import'; password: string; mnemonic: string }
  | { type: 'wallet:unlock'; password: string }
  | { type: 'wallet:lock' }
  | { type: 'wallet:getAccounts' }
  | { type: 'wallet:deriveAccount'; index: number };

// Response types -- background sends back
export type WalletResponse =
  | { type: 'wallet:created'; address: string; mnemonic: string }
  | { type: 'wallet:confirmed'; address: string }  // SEC-03: seed confirmed, vault persisted
  | { type: 'wallet:imported'; address: string }
  | { type: 'wallet:unlocked'; address: string }
  | { type: 'wallet:locked' }
  | { type: 'wallet:accounts'; accounts: DerivedAccount[] }
  | { type: 'wallet:derived'; account: DerivedAccount }
  | { type: 'wallet:error'; error: string };
```

1. **`src/features/wallet/crypto/mnemonic.ts`** -- BIP-39 mnemonic operations:
   - `generateMnemonic(strength?: 128 | 256): string` -- defaults to 128 (12 words). Uses `@scure/bip39` `generateMnemonic(wordlist, strength)`.
   - `isValidMnemonic(mnemonic: string): boolean` -- uses `validateMnemonic(mnemonic, wordlist)`.
   - `mnemonicToSeed(mnemonic: string): Promise<Uint8Array>` -- returns 64-byte seed. Uses `@scure/bip39` `mnemonicToSeed()`.
   - Import wordlist from `@scure/bip39/wordlists/english`.
   - **No logging of mnemonic values.** If a mnemonic is invalid, throw with generic message "Invalid mnemonic" -- never echo the input back.

2. **`src/features/wallet/crypto/hd.ts`** -- BIP-44 HD key derivation:
   - `deriveAccount(seed: Uint8Array, index: number): KeyPair` -- derives at `m/44'/60'/0'/0/${index}`. Uses `@scure/bip32` `HDKey.fromMasterSeed(seed).derive(path)`. Returns `{ privateKey, address, path }`.
   - `deriveAccounts(seed: Uint8Array, count: number): KeyPair[]` -- derives indices 0 through count-1.
   - Uses `privateKeyToAddress` from address.ts for the address field.
   - Throws if `account.privateKey` is null (should never happen but guard it).

3. **`src/features/wallet/crypto/address.ts`** -- Ethereum address from private key:
   - `privateKeyToAddress(privateKey: Uint8Array): string` -- secp256k1 getPublicKey (uncompressed, pass `false`!) -> slice off 04 prefix -> keccak256 -> last 20 bytes -> toChecksumAddress.
   - `toChecksumAddress(hex: string): string` -- EIP-55 mixed-case checksum. Input is 40-char hex WITHOUT 0x prefix. keccak256 the lowercase, uppercase chars where hash nibble >= 8, prepend 0x.
   - Import `secp256k1` from `@noble/curves/secp256k1`, `keccak_256` from `@noble/hashes/sha3`, `bytesToHex` from `@noble/hashes/utils`.
   - **CRITICAL:** `secp256k1.getPublicKey(privateKey, false)` -- the `false` means uncompressed 65-byte key. Default is compressed (33 bytes) which gives WRONG Ethereum addresses.

4. **`tests/crypto/mnemonic.test.ts`** -- BIP-39 test vectors:
